fetch(url)
    .then(response => response.json())
    .then(data => {
        const forecasts = data.list;
        const today = new Date();
        const threeDaysBack = new Date(today);
        threeDaysBack.setDate(today.getDate() - 3);
        const threeDaysForward = new Date(today);
        threeDaysForward.setDate(today.getDate() + 3);

        const pastForecasts = forecasts.filter(forecast => {
            const forecastDate = new Date(forecast.dt_txt);
            return forecastDate >= threeDaysBack && forecastDate < today;
        });

        const futureForecasts = forecasts.filter(forecast => {
            const forecastDate = new Date(forecast.dt_txt);
            return forecastDate > today && forecastDate <= threeDaysForward;
        });

        // Exibir os dados nas divs
        displayForecasts('past-forecasts', pastForecasts);
        displayForecasts('future-forecasts', futureForecasts);
    })
    .catch(error => console.error('Erro:', error));

function displayForecasts(divId, forecasts) {
    const div = document.getElementById(divId);
    div.innerHTML = '';
    forecasts.forEach(forecast => {
        const forecastDate = new Date(forecast.dt_txt);
        const forecastElement = document.createElement('div');
        forecastElement.innerHTML = `
            <p>${forecastDate.toLocaleDateString()} - ${forecast.weather[0].description}</p>
            <p>Temp: ${forecast.main.temp}Â°C</p>
        `;
        div.appendChild(forecastElement);
    });
}
